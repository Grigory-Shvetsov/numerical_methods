		\item \textit{–ö–∞–∫–∏–µ –∏–∑ —Ä–∞—Å—Å–º–æ—Ç—Ä–µ–Ω–Ω—ã—Ö —Å—Ö–µ–º —è–≤–ª—è—é—Ç—Å—è –∞–±—Å–æ–ª—é—Ç–Ω–æ —É—Å—Ç–æ–π—á–∏–≤—ã–º–∏? –ö–∞–∫–∞—è –∏–∑ —Ä–∞—Å—Å–º–æ—Ç—Ä–µ–Ω–Ω—ã—Ö —Å—Ö–µ–º –ø–æ–∑–≤–æ–ª—è–µ—Ç –≤–µ—Å—Ç–∏ —Ä–∞—Å—á–µ—Ç—ã —Å –±–æ–ª–µ–µ –∫—Ä—É–ø–Ω—ã–º —à–∞–≥–æ–º –ø–æ –≤—Ä–µ–º–µ–Ω–∏?}

		\]
		
		–ó–¥–µ—Å—å $\mu_{k, i} = \sin \sfrac{\pi k i h}l$, $k = \overline{1, n-1}$ --- —Å–æ–±—Å—Ç–≤–µ–Ω–Ω—ã–µ —Ñ—É–Ω–∫—Ü–∏–∏ —Ä–∞–∑–Ω–æ—Å—Ç–Ω–æ–π –ø—Ä–æ–∏–∑–≤–æ–¥–Ω–æ–π. –ü–æ–¥—Å—Ç–∞–≤–ª—è—è –≤ —Å—Ö–µ–º—É, –ø–æ–ª—É—á–∞–µ–º —Å–ª–µ–¥—É—é—â–µ–µ –≤—ã—Ä–∞–∂–µ–Ω–∏–µ:

			& \sum\limits_{k=1}^{n-1} \left(c\rho \dfrac{c_k^{j+1}-c_k^j}\tau\right) \mu_{k, i} = \sum\limits_{k=1}^{n-1} -\lambda_k^2 a(\sigma c_k^{j+1} + (1-\sigma) c_k^j) \mu_{k, i}.

		\]
		
		–ò–∑ —ç—Ç–æ–≥–æ —Å–ª–µ–¥—É–µ—Ç, —á—Ç–æ –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç —É $j$-–π –≥–∞—Ä–º–æ–Ω–∏–∫–∏ –º–µ–Ω—è–µ—Ç—Å—è –≤

		\]

		\]

		\]

		\item \textit{–ö–∞–∫–∏–µ –æ–≥—Ä–∞–Ω–∏—á–µ–Ω–∏—è –Ω–∞ $h,\,\tau$ –∏ $\sigma$ –Ω–∞–∫–ª–∞–¥—ã–≤–∞—é—Ç —É—Å–ª–æ–≤–∏—è —É—Å—Ç–æ–π—á–∏–≤–æ—Å—Ç–∏ –ø—Ä–æ–≥–æ–Ω–∫–∏?}

			A_i=\frac{\sigma}{h}a_i,\;	B_i=\frac{\sigma}{h}a_{i+1},\;C_i=\frac{\sigma}{h}a_i+\frac{\sigma}{h}a_{i+1}+c\rho\frac{h}{\tau},\\
